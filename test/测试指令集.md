  -H "Content-Type: application/json" \
  -d '{
    "username": "你的用户名",
    "password": "你的密码",
    "email": "邮箱"
  }' | jq

// 登陆获取jwt令牌

  -H "Content-Type: application/json" \
  -d '{
    "username": "suzuki",
    "password": "suzuki"
  }' | jq
