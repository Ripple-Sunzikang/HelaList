// 登陆获取jwt令牌
curl -X POST http://localhost:8080/api/user/login \
  -H "Content-Type: application/json" \
  -d '{
    "username": "suzuki",
    "password": "suzuki"
  }'

// 创建网盘数据到数据库
curl -X POST http://localhost:8080/api/storage/create \
  -H "Content-Type: application/json" \
  -d '{
    "mount_path": "/localwebdav",
    "driver": "webdav",
    "addition": "{\"address\":\"http://localhost:9999\",\"username\":\"suzuki\",\"password\":\"suzuki\",\"tls_insecure_skip_verify\":false}",
    "remark": "连接本地 WebDAV 服务器",
    "disabled": false,
    "order": 1,
    "cache_expiration": 3600,
    "status": "work"
  }'

// 查看网盘目录
curl -X GET "http://localhost:8080/api/fs/list/localwebdav" \
  -H "Content-Type: application/json" \
  -H "Authorization: Bearer 你的jwt令牌"
