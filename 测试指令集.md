// 注册新用户
curl -X POST http://localhost:8080/api/user/create \
  -H "Content-Type: application/json" \
  -d '{
    "username": "你的用户名",
    "password": "你的密码",
    "email": "邮箱"
  }' | jq

// 登陆获取jwt令牌
curl -X POST http://localhost:8080/api/user/login \
  -H "Content-Type: application/json" \
  -d '{
    "username": "suzuki",
    "password": "suzuki"
  }' | jq

// 创建网盘数据到数据库

    "remark": "连接本地 WebDAV 服务器",
    "disabled": false,
    "order": 1,
    "cache_expiration": 3600,

}

curl -X GET "http://localhost:8080/api/fs/list/localwebdav" \
 -H "Content-Type: application/json" \
 -H "Authorization: Bearer 你的jwt令牌" | jq
