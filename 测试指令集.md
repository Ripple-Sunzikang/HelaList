// 注册新用户
curl -X POST http://localhost:8080/api/user/create \
  -H "Content-Type: application/json" \
  -d '{
    "username": "你的用户名",
    "password": "你的密码",
    "email": "邮箱"
  }' | jq

// 登陆获取jwt令牌
curl -X POST http://localhost:8080/api/user/login \
  -H "Content-Type: application/json" \
  -d '{
    "username": "suzuki",
    "password": "suzuki"
  }' | jq

 -H "Content-Type: application/json" \
 -H "Authorization: Bearer <Your-Token>" | jq
